#include <stdio.h>

// 1. Verificador de Positividade [cite: 5, 6]
int verPositividade(int valor) {
    if (valor > 0) return 1;
    if (valor < 0) return -1;
    return 0;
}

// 2. Soma do Intervalo (excluindo extremos) [cite: 7, 8]
int somarIntervalo(int n1, int n2) {
    int soma = 0;
    // Garante que o loop percorra do menor para o maior
    int inicio = (n1 < n2) ? n1 : n2;
    int fim = (n1 > n2) ? n1 : n2;
    
    for (int i = inicio + 1; i < fim; i++) {
        soma += i;
    }
    return soma;
}

// 3. Divisíveis no Intervalo [cite: 9, 10]
int somarIntervaloDivisiveis(int a, int b, int c) {
    int soma = 0;
    for (int i = b; i <= c; i++) {
        if (i % a == 0) {
            soma += i;
        }
    }
    return soma;
}

// 4. Calculadora de Médias [cite: 11, 12, 13]
float calcularMediaTipo(float nota1, float nota2, float nota3, char tipo) {
    if (tipo == 'A') {
        return (nota1 + nota2 + nota3) / 3.0;
    } else if (tipo == 'P') {
        return ((nota1 * 2) + (nota2 * 4) + (nota3 * 5)) / 11.0;
    }
    return 0.0; // Retorno padrão caso char inválido
}

// 5. Média Ponderada com Pesos Livres [cite: 14, 15]
float calcularMediaPonderada(float n1, float n2, float n3, int p1, int p2, int p3) {
    return ((n1 * p1) + (n2 * p2) + (n3 * p3)) / (float)(p1 + p2 + p3);
}

// 6. Explorador de Extremos (Lendo N valores) [cite: 16, 17, 18]
void encontrarMinMax(int n) {
    int valor, maior, menor;
    if (n <= 0) return;

    printf("Digite o valor 1: ");
    scanf("%d", &valor);
    maior = menor = valor;

    for (int i = 1; i < n; i++) {
        printf("Digite o valor %d: ", i + 1);
        scanf("%d", &valor);
        if (valor > maior) maior = valor;
        if (valor < menor) menor = valor;
    }
    printf("Maior: %d, Menor: %d\n", maior, menor);
}

// 7. Soma dos Quadrados [cite: 19, 20]
int somaDosQuadrados(int a, int b, int c) {
    return (a * a) + (b * b) + (c * c);
}

// 8. Validador de Data [cite: 21, 22, 23]
int validarData(int dia, int mes, int ano) {
    if (ano < 0 || mes < 1 || mes > 12) return 0;
    
    int diasNoMes[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    
    // Verifica Bissexto
    if ((ano % 4 == 0 && ano % 100 != 0) || (ano % 400 == 0)) {
        diasNoMes[2] = 29;
    }

    if (dia >= 1 && dia <= diasNoMes[mes]) return 1;
    return 0;
}
